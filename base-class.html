Lab 14

Create a Selenium BaseClass.

Build one reusable parent class for browser launch, wait setup, URL loading, and driver cleanup.

Purpose

Why We Create BaseClass

A BaseClass keeps common Selenium setup in one place. Test classes can extend it and reuse the same driver, wait, browser launch, and close logic.

  • Avoid writing browser setup in every test class.
  • Keep driver initialization consistent.
  • Make teardown safe and repeatable.
  • Help framework files reuse the same starting point.
protected static WebDriver driver;

// 1. Browser Launch
public static WebDriver launchBrowser(String browserName) {

    try {

        if (browserName.equalsIgnoreCase("chrome")) {

            driver = new ChromeDriver();

        } else if (browserName.equalsIgnoreCase("edge")) {

            driver = new EdgeDriver();

        } else if (browserName.equalsIgnoreCase("firefox")) {

            driver = new FirefoxDriver();

        } else if (browserName.equalsIgnoreCase("safari")) {

            driver = new SafariDriver();

        } else {

            Assert.fail("Invalid Browser Name : " + browserName);
        }

        driver.manage().window().maximize();

    } catch (Exception e) {

        Assert.fail("Browser Launch Failed : " + e.getMessage());
    }

    return driver;
}
// 2. Close
public static void closeBrowser() {

    try {

        driver.close();

    } catch (Exception e) {

        Assert.fail("Error occurred during close : " + e.getMessage());
    }
}
// 3. Quit
public static void quitBrowser() {

    try {

        driver.quit();

    } catch (Exception e) {

        Assert.fail("Error occurred during quit : " + e.getMessage());
    }
}
